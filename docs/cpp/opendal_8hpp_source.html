211 public:
212 class ReaderStreamBuf : public std::streambuf {
213 public:
214 ReaderStreamBuf(Reader &&reader)
215 : reader_(std::move(reader)), buffer_start_pos_(0) {
216 setg(buffer_, buffer_, buffer_);
217 }
218
219 protected:
220 std::streamsize xsgetn(char *s, std::streamsize count) override {
221 std::streamsize total_read = 0;
222
223 while (total_read < count) {
224 if (gptr() < egptr()) {
225 std::streamsize available = egptr() - gptr();
226 std::streamsize to_copy = std::min(available, count - total_read);
227 std::memcpy(s + total_read, gptr(), to_copy);
228 gbump(to_copy);
229 total_read += to_copy;
230 continue;
231 }
232
233 if (underflow() == traits_type::eof()) break;
234 }
235
236 return total_read;
237 }
238
239 int_type underflow() override {
240 if (gptr() < egptr()) {
241 return traits_type::to_int_type(*gptr());
242 }
243
244 // Update the buffer start position to current reader position
245 buffer_start_pos_ += (egptr() - eback());
246
247 std::streamsize n = reader_.read(buffer_, sizeof(buffer_));
248 if (n <= 0) {
249 return traits_type::eof();
250 }
251
252 setg(buffer_, buffer_, buffer_ + n);
253 return traits_type::to_int_type(*gptr());
254 }
255
256 int_type uflow() override {
257 int_type result = underflow();
258 if (result != traits_type::eof()) {
259 gbump(1);
260 }
261 return result;
262 }
263
264 std::streampos seekoff(std::streamoff off, std::ios_base::seekdir dir,
265 std::ios_base::openmode which) override {
266 if (dir == std::ios_base::cur && off == 0) {
267 // tellg() case - return current position
268 return buffer_start_pos_ + (gptr() - eback());
269 }
270
271 // Actual seek operation
272 std::streampos new_pos = reader_.seek(off, dir);
273 if (new_pos != std::streampos(-1)) {
274 buffer_start_pos_ = new_pos;
275 setg(buffer_, buffer_, buffer_);
276 }
277 return new_pos;
278 }
279
280 std::streampos seekpos(std::streampos pos,
281 std::ios_base::openmode which) override {
282 return seekoff(pos, std::ios_base::beg, which);
283 }
284
285 private:
286 Reader reader_;
287 char buffer_[8192];
288 std::streampos buffer_start_pos_;
289 };
